import { ZuploContext, ZuploRequest } from "@zuplo/runtime";

export default async function (
  response: Response,
  request: ZuploRequest,
  context: ZuploContext,
) {
  // Clone the response to read the body without consuming the original response's body
  const clonedResponse = response.clone();
  const body = await clonedResponse.text();

  // Base64 encode the body
  // btoa is available in the edge runtime
  const signature = btoa(body);

  // Create a new Headers object from the original response headers
  const newHeaders = new Headers(response.headers);
  newHeaders.set("x-signature", signature);

  // Return a new response with the same body and the new headers
  return new Response(response.body, {
    status: response.status,
    statusText: response.statusText,
    headers: newHeaders,
  });
}
