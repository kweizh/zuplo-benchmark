```typescript
import { ZuploContext, ZuploRequest, environment } from "@zuplo/runtime";

export default async function (request: ZuploRequest, context: ZuploContext) {
  const upstreamUrl = environment.UPSTREAM_URL;
  const timeoutMs = parseInt(environment.UPSTREAM_TIMEOUT_MS, 10);

  try {
    const response = await fetch(upstreamUrl, {
      signal: AbortSignal.timeout(timeoutMs),
    });
    return response;
  } catch (err) {
    if (err.name === "TimeoutError" || err.name === "AbortError") {
      return new Response("Gateway Timeout", { status: 504 });
    }
    context.log.error(`Upstream request failed: ${err.message}`);
    return new Response(`Internal Server Error: ${err.message}`, { status: 500 });
  }
}
```

This implementation ensures that any request to `/proxy` will respect the strict timeout defined in the environment variables, providing a responsive gateway even when backends are slow or hanging.
