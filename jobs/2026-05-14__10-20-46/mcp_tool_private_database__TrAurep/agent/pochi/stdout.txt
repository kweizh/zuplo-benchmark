import { ZuploContext, ZuploRequest } from "zuplo";

export default async function (
  request: ZuploRequest,
  context: ZuploContext,
) {
  const body = await request.json();
  const region = body.region;

  if (!region) {
    return new Response("Missing region", { status: 400 });
  }
  
  const response = await context.invokeRoute("get-internal-sales", {
    method: "GET",
    searchParams: { region },
    headers: {
      authorization: "Bearer secret-internal-key"
    }
  });

  if (!response.ok) {
    return new Response("Error fetching sales: " + (await response.text()), { status: response.status });
  }

  const sales = await response.json();
  const totalSales = sales.reduce((acc: number, item: any) => acc + item.amount, 0);

  return {
    region,
    totalSales
  };
}
