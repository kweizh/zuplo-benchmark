### Implementation Details:

The handler uses the standard `fetch` API and the `ZuploContext` to log retry attempts. Since the request is a `GET`, it forwards the method and headers to the upstream service without needing to clone a request body.

```typescript
import { ZuploContext, ZuploRequest } from "@zuplo/runtime";

export default async function (
  request: ZuploRequest,
  context: ZuploContext,
) {
  const url = "http://127.0.0.1:8080/api/data";
  const maxRetries = 3;
  let lastResponse: Response;

  for (let i = 0; i <= maxRetries; i++) {
    try {
      lastResponse = await fetch(url, {
        method: request.method,
        headers: request.headers,
      });

      if (lastResponse.status !== 503) {
        return lastResponse;
      }

      context.log.info(`Attempt ${i + 1} returned 503. ${i < maxRetries ? "Retrying..." : "Max retries reached."}`);
    } catch (err) {
      context.log.error(`Error during fetch attempt ${i + 1}: ${err.message}`);
      throw err;
    }
  }

  return lastResponse;
}
```   
