import { ZuploContext, ZuploRequest } from "@zuplo/runtime";
import { MCPServer } from "@zuplo/mcp/server";
import { HttpStreamableTransport } from "@zuplo/mcp/transport/httpstreamable";
import { z } from "zod";

const server = new MCPServer({ name: "my-mcp", version: "1.0.0" });

server.addTool({
  name: "get_customer_summary",
  description: "Get customer summary",
  validator: z.object({ customerId: z.string() }),
  handler: async (args) => {
    return {
      content: [{ type: "text", text: `Summary for customer ${args.customerId}: VIP customer.` }]
    };
  }
});

export default async function (request: ZuploRequest, context: ZuploContext) {
  const transport = new HttpStreamableTransport(request, context);
  server.withTransport(transport);
  return transport.handleRequest();
}
EOF
