import { ZuploContext, ZuploRequest } from "@zuplo/runtime";

export default async function (request: ZuploRequest, context: ZuploContext) {
  // Read the JSON body
  const body = await request.json();
  
  // Simple JSON to XML converter for {"user": {"name": "Alice", "role": "admin"}}
  let xml = "";
  if (body && body.user) {
    xml = `<user><name>${body.user.name}</name><role>${body.user.role}</role></user>`;
  } else {
    xml = `<error>Invalid payload</error>`;
  }

  // Create a new request with the XML body
  const newRequest = new ZuploRequest(request, {
    body: xml
  });

  // Set the correct Content-Type header
  newRequest.headers.set("content-type", "application/xml");

  return newRequest;
}
